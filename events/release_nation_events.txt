country_event = {
	id = release_nation_events.60
	title = release_nation_events.60.t	# Freedom for ANG
	desc = release_nation_events.60.d	#
	picture = GFX_report_event_generic_sign_treaty3
	
	is_triggered_only = yes
	
	option = {
		name = release_nation_events.60.a
		if = {
			limit = { 
				controls_state = 540 
			}
			AN0 = { transfer_state = 540 }
		}
		if = {
			limit = { 
				has_country_flag = release_not_local_rulers
			}
			AN0 = { set_government_to_root = yes }
		}
		if = {
			limit = {
				has_country_flag = release_as_puppet
			}
			ROOT = { set_autonomy = { target = AN0 autonomy_state = autonomy_puppet } }
		}
		else_if = {
			limit = {
				has_country_flag = release_as_reichsprotectorate
			}
			ROOT = { set_autonomy = { target = AN0 autonomy_state = autonomy_reichsprotectorate } }
		}
		else_if = {
			limit = {
				has_country_flag = release_as_dominion
			}
			ROOT = { set_autonomy = { target = AN0 autonomy_state = autonomy_dominion } }
			AN0 = { add_opinion_modifier = { target = ROOT modifier = ROOT_released_us } }
		}
		else_if = {
			limit = {
				has_country_flag = release_as_imperial_protectorate
			}
			ROOT = { set_autonomy = { target = AN0 autonomy_state = autonomy_wtt_imperial_protectorate } }
		}
		else_if = {
			limit = {
				has_country_flag = release_as_imperial_associate
			}
			ROOT = { set_autonomy = { target = AN0 autonomy_state = autonomy_wtt_imperial_associate } }
		}
		else_if = {
			limit = {
				has_country_flag = release_as_supervised_state
			}
			ROOT = { set_autonomy = { target = AN0 autonomy_state = autonomy_supervised_state } }
		}
		else_if = {
			limit = {
				has_country_flag = release_as_satellite
			}
			ROOT = { set_autonomy = { target = AN0 autonomy_state = autonomy_satellite } }
		}
		else_if = {
			limit = {
				has_country_flag = release_as_reichskommissariat
			}
			ROOT = { set_autonomy = { target = AN0 autonomy_state = autonomy_reichskommissariat } }

		else_if = {
