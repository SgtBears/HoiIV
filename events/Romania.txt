# Soviet Union demands Bessarabia
country_event = {
	id = romania.1
	title = romania.1.t
	desc = romania.1.d
	picture = GFX_report_event_generic_read_write
	
	is_triggered_only = yes
	
	
	option = { # Accept demands
		name = romania.1.a
		ai_chance = {
			factor = 90
			modifier = { # Don't accept if in a faction that can protect them
				factor = 0
				OR = {
					any_other_country = {
						is_major = yes
						NOT = { tag = SO0 }
						OR = {
							is_in_faction_with = RO0
							has_guaranteed = RO0
						}
					}
					has_idea = anti_soviet_pact
				}
				is_historical_focus_on = no #Romania will still cede on historical
			}
		}
		SO0 = {
			country_event = { days = 1 id = soviet.4 }
		}
		set_country_flag = bessarabia_ceded
	}
	option = { # Refuse
		name = romania.1.b
		ai_chance = {
			base = 0
			modifier = {
				add = 10
				is_in_faction = yes
				NOT = { is_in_faction_with = SO0 }
			}
			modifier = {
				is_historical_focus_on = yes
				factor = 0
			}
		}
		RO0 = {
			SO0 = {
				country_event = { days = 1 id = soviet.5 }
				effect_tooltip = {
					create_wargoal = {
						type = take_state
						target = PREV
						generator = { 78 }
					}
				}
			}
		}
	}
}

# Seek an alliance - Since everyone is out to get us
country_event = {
	id = romania.6
	title = romania.6.t
	desc = romania.6.d
	picture = GFX_report_event_hitler_croatia_handshake
	
	trigger = {
		has_country_flag = bessarabia_ceded
		is_in_faction = no
		NOT = { GE0 = { is_in_faction_with = SO0 } }
		NOT = { has_war_with = GE0 }
		NOT = { has_war_with = SO0 }
		NOT = { has_dlc = "Death or Dishonor" }
	}
	
	mean_time_to_happen = { days = 7 }

	fire_only_once = yes

	option = { # Seek german alliance
		name = romania.6.a
		set_politics = {
			ruling_party = fascism
			elections_allowed = no
		}
		add_popularity = {
			ideology = fascism
			popularity = 0.35
		}
		GE0 = {
			ROOT = {
				add_opinion_modifier = { target = PREV modifier = ROM_seek_alliance_GER }
			}
		}
		add_ideas = ROM_revenge_against_soviet
		ai_chance = { 
			base = 2
			modifier = {
				factor = 0
				has_war_with = GE0
			}
		}
	}
	option = { # Seek soviet alliance
		name = romania.6.b
		set_politics = {
			ruling_party = communism
			elections_allowed = no
		}
		add_popularity = {

			popularity = 0.5
		}
		SO0 = {
			ROOT = {
				add_opinion_modifier = { target = PREV modifier = ROM_seek_alliance_SOV }
			}
		}
		add_ideas = ROM_revenge_against_germany
		ai_chance = {
			base = 1
			modifier = {
				factor = 0
				NOT = { has_war_with = GE0 }

			}
			modifier = {
				factor = 0
				is_historical_focus_on = yes
				NOT = { has_war_with = GE0 }
			}
		}
	}
	option = { # Dear democracy... NEVER GONNA GIVE YOU UP!
		name = romania.6.c
		ai_chance = {
			base = 0
		}
		
	}
}
